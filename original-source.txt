                    testsSection.classList.add('show');
                } else {
                    testsSection.style.display = 'none';
                    testsSection.classList.remove('show');
                }
            });
        });
        // ⭐⭐ الكود لفيديوهات الأنشطة ⭐⭐
        // إضافة حدث النقر لأزرار فيديوهات الأنشطة
        document.querySelectorAll('#activities .unit-btn').forEach(button => {
            button.addEventListener('click', function(e) {
                e.stopPropagation();
                
                // تبديل حالة الزر (نشط/غير نشط)
                this.classList.toggle('active');
                
                // البحث عن عنصر الفيديوهات التابع لهذا الزر
                const activitiesSection = this.nextElementSibling;
                
                // إذا كان الزر نشطاً، عرض الفيديوهات، وإلا إخفاؤها
                if (this.classList.contains('active')) {
                    activitiesSection.style.display = 'block';
                    activitiesSection.classList.add('show');
                } else {
                    activitiesSection.style.display = 'none';
                    activitiesSection.classList.remove('show');
                }
            });
        });
